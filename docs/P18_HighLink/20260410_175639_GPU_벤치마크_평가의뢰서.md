### 2-2. 추가 GPU 벤치마크 (아래 Python 스크립트를 직접 실행)

```python
"""GPU 전용 확장 벤치마크"""
import json
import time
from pathlib import Path

import cv2
import numpy as np
import torch
from ultralytics import YOLO

RESULT_DIR = Path("data/results")
RESULT_DIR.mkdir(parents=True, exist_ok=True)
SAMPLE_DIR = Path("data/samples")

# ── GPU 정보 수집 ──
gpu_info = {
    "cuda_available": torch.cuda.is_available(),
    "device_name": torch.cuda.get_device_name(0) if torch.cuda.is_available() else "N/A",
    "cuda_version": torch.version.cuda or "N/A",
    "torch_version": torch.__version__,
    "vram_total_gb": round(torch.cuda.get_device_properties(0).total_mem / 1024**3, 1) if torch.cuda.is_available() else 0,
}
print(f"GPU: {gpu_info['device_name']}, VRAM: {gpu_info['vram_total_gb']}GB, CUDA: {gpu_info['cuda_version']}")

# ── 테스트 매트릭스 ──
# 모델 x 해상도 조합별 FPS 측정
test_matrix = [
    {"model": "yolo11l.pt", "imgsz": 1280, "label": "YOLO11L@1280 (운영 기본)"},
    {"model": "yolo11l.pt", "imgsz": 640,  "label": "YOLO11L@640 (속도 우선)"},
    {"model": "yolo11s.pt", "imgsz": 1280, "label": "YOLO11S@1280 (엣지 후보)"},
    {"model": "yolo11s.pt", "imgsz": 640,  "label": "YOLO11S@640 (엣지 경량)"},
]

TARGET_CLASSES = [0, 1, 2, 3, 5, 7]  # person, bicycle, car, motorcycle, bus, truck

def benchmark_model(model_name, imgsz, video_path, max_frames=200):
    """단일 모델+해상도 벤치마크"""
    model = YOLO(model_name)
    
    cap = cv2.VideoCapture(str(video_path))
    if not cap.isOpened():
        return None
    
    w = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
    h = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
    
    # 워밍업 5프레임
    for _ in range(5):
        ret, frame = cap.read()
        if ret:
            model.predict(frame, conf=0.4, imgsz=imgsz, classes=TARGET_CLASSES,
                         device="cuda:0", verbose=False)
    cap.set(cv2.CAP_PROP_POS_FRAMES, 0)
    
    # VRAM 측정
    torch.cuda.reset_peak_memory_stats()
    
    frame_times = []
    det_counts = []
    frame_count = 0
    
    while frame_count < max_frames:
        ret, frame = cap.read()
        if not ret:
            cap.set(cv2.CAP_PROP_POS_FRAMES, 0)  # 루프
            ret, frame = cap.read()
            if not ret:
                break
        
        torch.cuda.synchronize()
        start = time.perf_counter()
        results = model.predict(frame, conf=0.4, iou=0.5, imgsz=imgsz,
                               classes=TARGET_CLASSES, device="cuda:0", verbose=False)
        torch.cuda.synchronize()
        elapsed = time.perf_counter() - start
        
        frame_times.append(elapsed)
        det_counts.append(len(results[0].boxes))
        frame_count += 1
    
    cap.release()
    
    vram_peak_gb = torch.cuda.max_memory_allocated() / 1024**3
    
    return {
        "frames_tested": frame_count,
        "resolution": f"{w}x{h}",
        "avg_fps": round(1.0 / np.mean(frame_times), 1),
        "avg_ms": round(np.mean(frame_times) * 1000, 1),
        "p95_ms": round(np.percentile(frame_times, 95) * 1000, 1),
        "p99_ms": round(np.percentile(frame_times, 99) * 1000, 1),
        "min_ms": round(min(frame_times) * 1000, 1),
        "max_ms": round(max(frame_times) * 1000, 1),
        "avg_detections": round(np.mean(det_counts), 1),
        "vram_peak_gb": round(vram_peak_gb, 2),
    }

# ── 비디오 소스 확인 ──
video_path = SAMPLE_DIR / "highway_sample.mp4"
if not video_path.exists():
    # 샘플 없으면 sample_test.py가 다운로드한 이미지로 합성 비디오 생성
    print("highway_sample.mp4 없음 — sample_test.py를 먼저 실행해 주세요.")
    print("또는 yt-dlp로 다운로드: pip install yt-dlp")
    print('yt-dlp -f "136+140" --download-sections "*0:00-0:30" -o "data/samples/highway_sample.mp4" "https://www.youtube.com/watch?v=MNn9qKG2UFI"')
    exit(1)

# ── 벤치마크 실행 ──
results = {"gpu_info": gpu_info, "tests": {}}

for test in test_matrix:
    print(f"\n{'='*50}")
    print(f"테스트: {test['label']}")
    print(f"{'='*50}")
    
    result = benchmark_model(test["model"], test["imgsz"], video_path)
    if result:
        results["tests"][test["label"]] = result
        print(f"  FPS: {result['avg_fps']} (avg {result['avg_ms']}ms, p95 {result['p95_ms']}ms)")
        print(f"  VRAM: {result['vram_peak_gb']}GB")
        print(f"  평균 감지: {result['avg_detections']}/프레임")

# ── 멀티스트림 시뮬레이션 (2스트림) ──
print(f"\n{'='*50}")
print("멀티스트림 테스트: YOLO11L@1280 x 2스트림 (배치 추론)")
print(f"{'='*50}")

model = YOLO("yolo11l.pt")
cap = cv2.VideoCapture(str(video_path))

# 워밍업
for _ in range(5):
    ret, frame = cap.read()
    if ret:
        model.predict([frame, frame], conf=0.4, imgsz=1280, classes=TARGET_CLASSES,
                     device="cuda:0", verbose=False, batch=2)
cap.set(cv2.CAP_PROP_POS_FRAMES, 0)

torch.cuda.reset_peak_memory_stats()
batch_times = []
for i in range(100):
    ret, frame = cap.read()
    if not ret:
        cap.set(cv2.CAP_PROP_POS_FRAMES, 0)
        ret, frame = cap.read()
    
    torch.cuda.synchronize()
    start = time.perf_counter()
    model.predict([frame, frame], conf=0.4, iou=0.5, imgsz=1280,
                 classes=TARGET_CLASSES, device="cuda:0", verbose=False, batch=2)
    torch.cuda.synchronize()
    batch_times.append(time.perf_counter() - start)

cap.release()

multi_result = {
    "streams": 2,
    "total_fps": round(2.0 / np.mean(batch_times), 1),  # 2프레임/배치
    "per_stream_fps": round(1.0 / np.mean(batch_times), 1),
    "avg_ms_per_batch": round(np.mean(batch_times) * 1000, 1),
    "p95_ms_per_batch": round(np.percentile(batch_times, 95) * 1000, 1),
    "vram_peak_gb": round(torch.cuda.max_memory_allocated() / 1024**3, 2),
}
results["multi_stream"] = multi_result
print(f"  총 FPS: {multi_result['total_fps']} ({multi_result['per_stream_fps']} fps/스트림)")
print(f"  VRAM: {multi_result['vram_peak_gb']}GB")

# ── JSON 저장 ──
report_path = RESULT_DIR / "gpu_benchmark_raw.json"
with open(report_path, "w", encoding="utf-8") as f:
    json.dump(results, f, ensure_ascii=False, indent=2)
print(f"\n원시 결과 저장: {report_path}")
```

위 코드를 `scripts/gpu_benchmark.py`로 저장하고 실행:
```bash
python scripts/gpu_benchmark.py
```
